🔗 Connect to Google Sheets

Do this once. After setup, every case syncs instantly to your Sheet.

1
Go to sheets.google.com → create a new sheet called DSA Cases
2
Click Extensions → Apps Script
3
Delete existing code and paste the script below, then click Save
function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Sheet1") || ss.getSheets()[0]; if (sheet.getLastRow() === 0) { sheet.appendRow([ "Timestamp","Patient ID","Age","Gender","Date", "Indication","Procedure","Territory", "Devices","Device Specs","Outcome", "Complications","Teaching","HDD","Path","Notes" ]); } var d = JSON.parse(e.postData.contents); sheet.appendRow([ new Date().toLocaleString(), d.pid, d.age, d.gender, d.date, d.indication, d.proc, d.territory, d.devices, d.devicespec, d.outcome, d.comp, d.teach, d.hdd, d.path, d.notes ]); return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService .createTextOutput("DSA Logger OK") .setMimeType(ContentService.MimeType.TEXT); }
4
Click Deploy → New Deployment
Type: Web App · Who has access: Anyone
